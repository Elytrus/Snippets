struct disjoint{
	int set[MAX], size;

	disjoint(int sz){
		size = sz;
		for(int i = 0; i < size; i++) set[i] = i;
	}

	int root(int val){
		if(set[val] == val) return val;
		return set[val] = root(set[val]);
	}

	void unite(int v1, int v2){ // v1 becomes the "parent" of v2
		set[root(v2)] = root(v1);
	}

	bool same(int v1, int v2){
		return root(v1) == root(v2);
	}
};
